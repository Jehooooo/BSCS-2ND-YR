// Enrollment Class Definition
class Enrollment {
    constructor(studentName, yearLevel, course, studentId) {
        this.studentName = studentName;
        this.yearLevel = yearLevel;
        this.course = course;
        this.subjects = [];
        this.studentId = studentId;
        this.status = "Not Enrolled";
    }

    // Method to enlist a subject for enrollment
    subjectEnlistment(subject) {
        if (!this.subjects.includes(subject)) {
            this.subjects.push(subject);
            return `${subject} has been added to your enlistment list.`;
        }
        return `${subject} is already in your enlistment list.`;
    }

    // Method to enroll in a subject
    enrollSubject(subject) {
        if (this.subjects.includes(subject)) {
            this.status = "Enrolled";
            return `${this.studentName} has successfully enrolled in ${subject}.`;
        }
        return `${subject} is not in your enlistment list. Please enlist first.`;
    }

    // Method to update enrollment status
    updateStatus(newStatus) {
        this.status = newStatus;
        return `Enrollment status updated to: ${newStatus}`;
    }

    // Method to display current enrollment information
    displayEnrollment() {
        return `
            Student Name: ${this.studentName}
            Student ID: ${this.studentId}
            Year Level: ${this.yearLevel}
            Course: ${this.course}
            Subjects: ${this.subjects.join(", ")}
            Status: ${this.status}
        `;
    }
}

// Creating an instance of Enrollment for Jehosue
const jehosue = new Enrollment(
    "Jehosue First Yr",
    "First Year",
    "Computer Science",
    "CS2023-001"
